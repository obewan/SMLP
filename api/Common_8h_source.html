123 // Trim CRLF
124 inline std::string trimCRLF(const std::string &str) {
125  std::string trimmed(str);
126  while (!trimmed.empty() &&
127  (trimmed.front() == '\r' || trimmed.front() == '\n')) {
128  trimmed.erase(trimmed.begin());
129  }
130  while (!trimmed.empty() &&
131  (trimmed.back() == '\r' || trimmed.back() == '\n')) {
132  trimmed.pop_back();
133  }
134  return trimmed;
135 }
